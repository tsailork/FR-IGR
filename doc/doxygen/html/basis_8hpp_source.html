39inline double legendre(int n, double x) {
40 if (n == 0) return 1.0;
41 if (n == 1) return x;
42 double Lm2 = 1.0, Lm1 = x, L = 0.0;
43 for (int k = 2; k <= n; ++k) {
44 L = ((2.0 * k - 1.0) * x * Lm1 - (k - 1.0) * Lm2) / k;
45 Lm2 = Lm1;
46 Lm1 = L;
47 }
48 return L;
49}

63inline double legendre_prime(int n, double x) {
64 if (n == 0) return 0.0;
65 if (std::abs(x) >= 1.0 - 1e-12) {
66 double sign = (x < 0 && ((n + 1) % 2 != 0)) ? -1.0 : 1.0;
67 return sign * 0.5 * n * (n + 1);
68 }
69 return (n * (x * legendre(n, x) - legendre(n - 1, x))) / (x * x - 1.0);
70}
