465{
466 sig_neigh = 0.0;
467 for (int v = 0; v < 4; ++v) neigh_state[v] = 0.0;
468
469 const NeighborInfo& ni = is_right ? b.ni_r : b.ni_l;
470
471 if (!is_right) {
472 // ---- Left interface ----

474 if (ni.id != -1) {
475 const Block& nb = blocks[ni.id];
476 int nex = (ni.face == 'L') ? 0 : nb.nx - 1;
477 const double* weights = (ni.face == 'L') ? basis.l_L.data() : basis.l_R.data();
478 for (int k = 0; k < p.N_PTS; ++k) {
479 for (int v = 0; v < 4; ++v)
480 neigh_state[v] += nb.U(v, ey, nex, iy, k) * weights[k];
481 sig_neigh += nb.sigma_field[nb.get_flat_idx(ey, nex, iy, k, p.N_PTS)] * weights[k];
482 }
483
484 } else if (ni.is_noslip_wall || ni.is_moving_wall) {
485 // Viscous wall: tangential direction for L/R walls is y
