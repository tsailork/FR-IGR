40inline double pressure(double rho, double rhou, double rhov, double E,
41 double gamma) {
42 return (gamma - 1.0) * (E - 0.5 * (rhou*rhou + rhov*rhov) / rho);

55inline double specific_entropy(double rho, double rhou, double rhov,
56 double E, double gamma) {
57 double p = pressure(rho, rhou, rhov, E, gamma);
58 return p / std::pow(rho, gamma);

76inline double pressure_at_theta(double theta,
77 double r, double ru, double rv, double E,
78 double r_avg, double ru_avg, double rv_avg,
79 double E_avg, double gamma) {
80 double rt = theta * r + (1.0 - theta) * r_avg;
81 double rut = theta * ru + (1.0 - theta) * ru_avg;
82 double rvt = theta * rv + (1.0 - theta) * rv_avg;
83 double Et = theta * E + (1.0 - theta) * E_avg;
84 return (gamma - 1.0) * (Et - 0.5 * (rut*rut + rvt*rvt) / rt);
85}

102inline double entropy_at_theta(double theta,
103 double r, double ru, double rv, double E,
104 double r_avg, double ru_avg, double rv_avg,
105 double E_avg, double gamma) {
106 double rt = theta * r + (1.0 - theta) * r_avg;
107 double rut = theta * ru + (1.0 - theta) * ru_avg;
108 double rvt = theta * rv + (1.0 - theta) * rv_avg;
109 double Et = theta * E + (1.0 - theta) * E_avg;
110 double pt = (gamma - 1.0) * (Et - 0.5 * (rut*rut + rvt*rvt) / rt);
111 return pt / std::pow(rt, gamma);
112}

134inline double bisect_for_theta(
135 double r, double ru, double rv, double E,
136 double r_avg, double ru_avg, double rv_avg, double E_avg,
137 double gamma, double target, bool is_pressure, int n_iter = 30)
138{
139 double lo = 0.0, hi = 1.0;
140 for (int iter = 0; iter < n_iter; ++iter) {
141 double mid = 0.5 * (lo + hi);
142 double val = is_pressure
143 ? pressure_at_theta(mid, r, ru, rv, E, r_avg, ru_avg, rv_avg, E_avg, gamma)
144 : entropy_at_theta(mid, r, ru, rv, E, r_avg, ru_avg, rv_avg, E_avg, gamma);
145 if (val >= target) lo = mid;
146 else hi = mid;
147 }
148 // Safety pullback to ensure strictly admissible bounds in the presence of float variations
149 return lo * (1.0 - 1e-8);
150}

173inline int extrapolate_face_values(const State& U, const Basis& basis,
174 int ey, int ex,
175 double face_pts[][4]) {
176 int count = 0;
177
178 // Left face (ξ = −1): for each iy, sum over ix with l_L[ix]
179 for (int iy = 0; iy < U.npts; ++iy) {
180 for (int v = 0; v < 4; ++v) face_pts[count][v] = 0.0;
181 for (int ix = 0; ix < U.npts; ++ix)
182 for (int v = 0; v < 4; ++v)
183 face_pts[count][v] += U(v, ey, ex, iy, ix) * basis.l_L[ix];
184 ++count;
185 }
186 // Right face (ξ = +1): for each iy, sum over ix with l_R[ix]
187 for (int iy = 0; iy < U.npts; ++iy) {
188 for (int v = 0; v < 4; ++v) face_pts[count][v] = 0.0;
189 for (int ix = 0; ix < U.npts; ++ix)
190 for (int v = 0; v < 4; ++v)
191 face_pts[count][v] += U(v, ey, ex, iy, ix) * basis.l_R[ix];
192 ++count;
193 }
194 // Bottom face (η = −1): for each ix, sum over iy with l_L[iy]
195 for (int ix = 0; ix < U.npts; ++ix) {
196 for (int v = 0; v < 4; ++v) face_pts[count][v] = 0.0;
197 for (int iy = 0; iy < U.npts; ++iy)
198 for (int v = 0; v < 4; ++v)
199 face_pts[count][v] += U(v, ey, ex, iy, ix) * basis.l_L[iy];
200 ++count;
201 }
202 // Top face (η = +1): for each ix, sum over iy with l_R[iy]
203 for (int ix = 0; ix < U.npts; ++ix) {
204 for (int v = 0; v < 4; ++v) face_pts[count][v] = 0.0;
205 for (int iy = 0; iy < U.npts; ++iy)
206 for (int v = 0; v < 4; ++v)
207 face_pts[count][v] += U(v, ey, ex, iy, ix) * basis.l_R[iy];
208 ++count;
209 }
210 return count;
211}

225inline void compute_cell_average(const State& U, const Basis& basis,
226 int ey, int ex,
227 double& r_avg, double& ru_avg,
228 double& rv_avg, double& E_avg) {
229 r_avg = ru_avg = rv_avg = E_avg = 0.0;
230 for (int iy = 0; iy < U.npts; ++iy)
231 for (int ix = 0; ix < U.npts; ++ix) {
232 double w = (basis.w[iy] * 0.5) * (basis.w[ix] * 0.5);
233 r_avg += w * U(0, ey, ex, iy, ix);
234 ru_avg += w * U(1, ey, ex, iy, ix);
235 rv_avg += w * U(2, ey, ex, iy, ix);
236 E_avg += w * U(3, ey, ex, iy, ix);
237 }
238}

249inline double min_entropy_in_cell(const State& U, const Parameters& p,
250 int ey, int ex) {
251 double s_min = 1e30;
252 for (int iy = 0; iy < U.npts; ++iy)
253 for (int ix = 0; ix < U.npts; ++ix) {
254 double r = std::max(1e-14, U(0, ey, ex, iy, ix));
255 double ru = U(1, ey, ex, iy, ix);
256 double rv = U(2, ey, ex, iy, ix);
257 double E = U(3, ey, ex, iy, ix);
258 s_min = std::min(s_min, specific_entropy(r, ru, rv, E, p.GAMMA));
259 }
260 return s_min;
261}
