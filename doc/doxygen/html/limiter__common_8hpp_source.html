27inline double pressure(double rho, double rhou, double rhov, double E,
28 double gamma) {
29 return (gamma - 1.0) * (E - 0.5 * (rhou*rhou + rhov*rhov) / rho);
30}

35inline double specific_entropy(double rho, double rhou, double rhov,
36 double E, double gamma) {
37 double p = pressure(rho, rhou, rhov, E, gamma);
38 return p / std::pow(rho, gamma);
39}

44inline double pressure_at_theta(double theta,
45 double r, double ru, double rv, double E,
46 double r_avg, double ru_avg, double rv_avg,
47 double E_avg, double gamma) {
48 double rt = theta * r + (1.0 - theta) * r_avg;
49 double rut = theta * ru + (1.0 - theta) * ru_avg;
50 double rvt = theta * rv + (1.0 - theta) * rv_avg;
51 double Et = theta * E + (1.0 - theta) * E_avg;
52 return (gamma - 1.0) * (Et - 0.5 * (rut*rut + rvt*rvt) / rt);
53}

58inline double entropy_at_theta(double theta,
59 double r, double ru, double rv, double E,
60 double r_avg, double ru_avg, double rv_avg,
61 double E_avg, double gamma) {
62 double rt = theta * r + (1.0 - theta) * r_avg;
63 double rut = theta * ru + (1.0 - theta) * ru_avg;
64 double rvt = theta * rv + (1.0 - theta) * rv_avg;
65 double Et = theta * E + (1.0 - theta) * E_avg;
66 double pt = (gamma - 1.0) * (Et - 0.5 * (rut*rut + rvt*rvt) / rt);
67 return pt / std::pow(rt, gamma);
68}

73inline double bisect_for_theta(
74 double r, double ru, double rv, double E,
75 double r_avg, double ru_avg, double rv_avg, double E_avg,
76 double gamma, double target, bool is_pressure, int n_iter = 30)
77{
78 double lo = 0.0, hi = 1.0;
79 for (int iter = 0; iter < n_iter; ++iter) {
80 double mid = 0.5 * (lo + hi);
81 double val = is_pressure
82 ? pressure_at_theta(mid, r, ru, rv, E, r_avg, ru_avg, rv_avg, E_avg, gamma)
83 : entropy_at_theta(mid, r, ru, rv, E, r_avg, ru_avg, rv_avg, E_avg, gamma);
84 if (val >= target) lo = mid;
85 else hi = mid;
86 }
87 return lo * (1.0 - 1e-8);
88}

98inline int extrapolate_face_values(const Cell& c, const Basis& basis,
99 double face_pts[][4], int npts) {
100 int count = 0;
101
102 // Left face (ξ = −1): for each iy, sum over ix with l_L[ix]
103 for (int iy = 0; iy < npts; ++iy) {
104 for (int v = 0; v < 4; ++v) face_pts[count][v] = 0.0;
105 for (int ix = 0; ix < npts; ++ix)
106 for (int v = 0; v < 4; ++v)
107 face_pts[count][v] += c.get_U(v, iy, ix, npts) * basis.l_L[ix];
108 ++count;
109 }
110 // Right face (ξ = +1): for each iy, sum over ix with l_R[ix]
111 for (int iy = 0; iy < npts; ++iy) {
112 for (int v = 0; v < 4; ++v) face_pts[count][v] = 0.0;
113 for (int ix = 0; ix < npts; ++ix)
114 for (int v = 0; v < 4; ++v)
115 face_pts[count][v] += c.get_U(v, iy, ix, npts) * basis.l_R[ix];
116 ++count;
117 }
118 // Bottom face (η = −1): for each ix, sum over iy with l_L[iy]
119 for (int ix = 0; ix < npts; ++ix) {
120 for (int v = 0; v < 4; ++v) face_pts[count][v] = 0.0;
121 for (int iy = 0; iy < npts; ++iy)
122 for (int v = 0; v < 4; ++v)
123 face_pts[count][v] += c.get_U(v, iy, ix, npts) * basis.l_L[iy];
124 ++count;
125 }
126 // Top face (η = +1): for each ix, sum over iy with l_R[iy]
127 for (int ix = 0; ix < npts; ++ix) {
128 for (int v = 0; v < 4; ++v) face_pts[count][v] = 0.0;
129 for (int iy = 0; iy < npts; ++iy)
130 for (int v = 0; v < 4; ++v)
131 face_pts[count][v] += c.get_U(v, iy, ix, npts) * basis.l_R[iy];
132 ++count;
133 }
134 return count;
135}

140inline void compute_cell_average(const Cell& c, const Basis& basis,
141 double& r_avg, double& ru_avg,
142 double& rv_avg, double& E_avg, int npts) {
143 r_avg = ru_avg = rv_avg = E_avg = 0.0;
144 for (int iy = 0; iy < npts; ++iy)
145 for (int ix = 0; ix < npts; ++ix) {
146 double w = (basis.w[iy] * 0.5) * (basis.w[ix] * 0.5);
147 r_avg += w * c.get_U(0, iy, ix, npts);
148 ru_avg += w * c.get_U(1, iy, ix, npts);
149 rv_avg += w * c.get_U(2, iy, ix, npts);
150 E_avg += w * c.get_U(3, iy, ix, npts);
151 }
152}

157inline double min_entropy_in_cell(const Cell& c, const Parameters& p, int npts) {
158 double s_min = 1e30;
159 for (int iy = 0; iy < npts; ++iy)
160 for (int ix = 0; ix < npts; ++ix) {
161 double r = std::max(1e-14, c.get_U(0, iy, ix, npts));
162 double ru = c.get_U(1, iy, ix, npts);
163 double rv = c.get_U(2, iy, ix, npts);
164 double E = c.get_U(3, iy, ix, npts);
165 s_min = std::min(s_min, specific_entropy(r, ru, rv, E, p.GAMMA));
166 }
167 return s_min;
168}
