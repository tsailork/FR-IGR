# Example inputs.dat configuration for FR-IGR
# This file uses standard INI format. Comments start with #.

[Physics]

[Solver]

[NavierStokes]

[Regularization]

ENABLE_IGR = false

# ALPHA_SCALE: Viscosity scaling coefficient. Higher = more diffusion.
ALPHA_SCALE = 0.5

# IGR_GRADIENT_TYPE: "LOCAL" (interior points only) or "CORRECTED" (BR2 interface lifting)
IGR_GRADIENT_TYPE = LOCAL

# IGR_TYPE: "ELLIPTIC" (implicit ADI solve) or "PARABOLIC" (explicit evolution)
IGR_TYPE = PARABOLIC

# IGR_TAU_R: Relaxation time for parabolic mode.
IGR_TAU_R = 0.1

IGR_BR2_ETA = 1.0

[Stabilization]

# ENABLE_ENTROPY_LIMITER: Prevents unphysical entropy undershoots.

[ImmersedBoundary]

ENABLE_IB = false

# IB_METHOD: Integration type. Options: 
#   - ANALYTICAL: Exact semi-implicit fractional-step integration (unconditionally stable for any eta).
#   - EXPLICIT: Adds VPM source terms directly to the RHS (requires dt <= eta for stability).
IB_METHOD = ANALYTICAL

# IB_SHAPE: Geometry definition. Supported: "CIRCLE", "NACA", "MULTI"
#   - CIRCLE: Uses standard single cylinder geometry.
#   - NACA: Uses a standard 4-digit airfoil geometry.
#   - MULTI: Allows user-defined lists of convex quadrilaterals & lines/parabolas.
IB_SHAPE = CIRCLE

# IB_NACA_CODE: Standard 4-digit NACA airfoil code (e.g. "0012", "2412"). Used if IB_SHAPE = NACA.
IB_NACA_CODE = 0012

# IB_AOA: Airfoil angle of attack (AOA) in degrees. Used if IB_SHAPE = NACA.
IB_AOA = 0.0

# Geometry dimensions of the cylinder/airfoil (Used if IB_SHAPE = CIRCLE or NACA)
# For CIRCLE: Center coordinate (IB_CENTER_X, IB_CENTER_Y) and Radius (IB_RADIUS).
# For NACA: Leading edge coordinate (IB_CENTER_X, IB_CENTER_Y) and Chord length (IB_RADIUS).
IB_CENTER_X = 2.0
IB_CENTER_Y = 1.0
IB_RADIUS = 0.25

# --- MULTI Shape Options (Used if IB_SHAPE = MULTI) ---
# IB_Q_TIME_MAP: Piecewise-linear checkpoints for time parameter q(t) (format: t_0:q_0 t_1:q_1 ...).
# Useful for prescribing moving/morphing geometries. Default is 0.0:0.0.
# Example: 0.0:0.0 1.0:0.0 2.0:1.0 (remains at q=0 from t=0 to 1, then linearly increases to q=1 at t=2)
IB_Q_TIME_MAP = 0.0:0.0 1.0:0.0 2.0:1.0

# IB_NUM_QUADS: Number of convex quadrilaterals to mask out.
IB_NUM_QUADS = 0
# IB_QUAD_i: 8 coordinates representing ordered vertices (x1 y1 x2 y2 x3 y3 x4 y4) for quadrilateral i.
# IB_QUAD_0 = 0.2 0.2 0.4 0.2 0.4 0.4 0.2 0.4

# IB_NUM_POLYS: Number of lines/parabolas to mask out.
IB_NUM_POLYS = 0
# IB_POLY_i: Defines curve f(xi) = a*xi^2 + b*xi + c where coefficients interpolate linearly with q(t).
# Format: DIR a0 b0 c0 a1 b1 c1 SIDE
#   - DIR: 'Y' for vertical offset f(x) = y, 'X' for horizontal offset f(y) = x
#   - a0, b0, c0: Coefficients [a, b, c] at state q = 0
#   - a1, b1, c1: Coefficients [a, b, c] at state q = 1
#   - SIDE: Which side of the curve to mask out: "ABOVE", "BELOW" (for DIR=Y) or "LEFT", "RIGHT" (for DIR=X)
# Example: Y 0.0 0.0 0.2 0.0 0.0 0.4 BELOW
#   (defines a flat horizontal line moving vertically from y=0.2 at q=0 to y=0.4 at q=1, masking below it)
# IB_POLY_0 = Y 0.0 0.0 0.2 0.0 0.0 0.4 BELOW

# IB_PENALIZATION_ETA: Solid permeability parameter (eta). Smaller = more rigid/impermeable.
IB_PENALIZATION_ETA = 1e-7

# Solid boundary velocities (static)
IB_VELOCITY_X = 0.0
IB_VELOCITY_Y = 0.0

# IB_THERMAL_TYPE: Solid thermal boundary condition. Options:
#   - ADIABATIC: Penalizes velocity only, leaving internal pressure/temperature adiabatic.
#   - ISOTHERMAL: Penalizes velocity and forces solid temperature to IB_TEMPERATURE.
IB_THERMAL_TYPE = ADIABATIC

# Target wall temperature (used if IB_THERMAL_TYPE = ISOTHERMAL)
IB_TEMPERATURE = 1.0

# IB_SHARP: Use a sharp solid boundary interface mask (true) or regularized Heaviside (false).
# Note: False is highly recommended for high-order methods to prevent Gibbs oscillations.
IB_SHARP = false

# IB_SMOOTH_WIDTH: Half-width thickness of the smoothed Heaviside boundary interface

IB_SMOOTH_WIDTH = 1.5

# DOMAIN GRID SYNTAX (domain.grid)
################################################################################
# The simulation domain is defined in domain.grid using multiple [BlockN] 
# sections. Each block is a Cartesian grid with its own resolution and BCs.
#
# Example [Block0] configuration:
# [Block0]
# N_ELEM_X = 20           # Number of elements in X direction
# N_ELEM_Y = 20           # Number of elements in Y direction
# X_MIN = 0.0             # Physical X coordinate of the left face
# X_MAX = 1.0             # Physical X coordinate of the right face
# Y_MIN = 0.0             # Physical Y coordinate of the bottom face
# Y_MAX = 1.0             # Physical Y coordinate of the top face
# 
# BC_L = TRANSMISSIVE    # Boundary Condition for Left face
# BC_R = 1:L             # Connects to Block 1's Left face
# BC_B = 0:T             # Connects to Block 0's Top face (self-periodic)
# BC_T = 0:B             # Connects to Block 0's Bottom face (self-periodic)
#
# Available BCs:
#   - OUTFLOW_SUPERSONIC: Simple copy-out/ghost-state condition.
#   - WALL: Reflective slip-wall (normal velocity negated).
#   - WALL_NOSLIP: Adiabatic viscous no-slip wall (u=v=0, dT/dn=0).
#   - WALL_NOSLIP:T: Isothermal viscous no-slip wall at temperature T.
#   - WALL_MOVING:V: Adiabatic moving wall with tangential velocity V.
#   - WALL_MOVING:V:T: Isothermal moving wall with velocity V and temperature T.
#   - INFLOW_SUPERSONIC:rho:u:v:p: Enforces the given reference state.
#   - CHARACTERISTIC:rho:u:v:p: Riemann invariant based subsonic inflow/outflow using the reference state.
#   - TOTAL_PRESSURE_COMP:Pt: Subsonic backpressure boundary scaling the state to match target compressible total pressure Pt.
#   - TOTAL_PRESSURE_INCOMP:Pt: Subsonic backpressure boundary using Bernoulli's incompressible equation to match target total pressure Pt.
#   - STATIC_PRESSURE:P: Subsonic backpressure boundary enforcing the target static pressure P while copying interior density and velocity.
#   - ID:FACE: Connects to the specified face of neighbor Block ID.
#     Example: '2:T' means "Connect to the TOP face of Block 2".
#     Note: For periodic boundaries, simply map opposite faces to each other 
#     (e.g., Block 0 Left connects to Block 0 Right using '0:R').
