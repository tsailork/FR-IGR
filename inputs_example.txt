# Example inputs.dat configuration for FR-IGR
# This file uses standard INI format. Comments start with #.

[Physics]
# GAMMA: Ratio of specific heats (1.4 for air)
GAMMA = 1.4

# IC_TYPE: Initial Condition. Options: RIEMANN_2D_C3, BLAST, SINE_WAVE, UNIFORM
IC_TYPE = BLAST

[Solver]
# P_DEG: Polynomial degree of elements. 0 = Finite Volume, 1 = Linear, 2 = Quadratic
P_DEG = 1

# CFL: Courant-Friedrichs-Lewy number for time step scaling
CFL = 1.0

# T_FINAL: Simulation end time
T_FINAL = 0.3

# NUM_THREADS: Number of OpenMP threads to use
NUM_THREADS = 4

[Regularization]
# ENABLE_IGR: Toggle Isotropic Gradient Regularization (artificial viscosity)
ENABLE_IGR = false

# ALPHA_SCALE: Viscosity scaling coefficient. Higher = more diffusion.
ALPHA_SCALE = 0.5

# IGR_GRADIENT_TYPE: "LOCAL" (interior points only) or "CORRECTED" (BR2 interface lifting)
IGR_GRADIENT_TYPE = LOCAL

# IGR_TYPE: "ELLIPTIC" (implicit ADI solve) or "PARABOLIC" (explicit evolution)
IGR_TYPE = PARABOLIC

# IGR_TAU_R: Relaxation time for parabolic mode.
IGR_TAU_R = 0.1

# IGR_BR2_ETA: BR2 penalty parameter for stability. Usually 1.0 to 3.0.
IGR_BR2_ETA = 1.0

# IGR_SUB_ITERS: Number of parabolic sub-iterations per flow step.
IGR_SUB_ITERS = 1

[Stabilization]
# ENABLE_POS_LIMITER: Zhang-Shu density and pressure positivity limiter.
ENABLE_POS_LIMITER = true

# POS_LIMITER_EPS: Minimum value floor for density and pressure.
POS_LIMITER_EPS = 1e-10

# ENABLE_ENTROPY_LIMITER: Prevents unphysical entropy undershoots.
ENABLE_ENTROPY_LIMITER = true

[IO]
# OUTPUT_DT: Time interval between full VTK 3D snapshots.
OUTPUT_DT = 0.01

# RESIDUAL_INTERVAL: Time interval between L2 residual tracking outputs to residuals.dat
RESIDUAL_INTERVAL = 0.005

# PROBE_INTERVAL: Time interval between point probe tracking to probe.csv
PROBE_INTERVAL = 0.005

# PRINT_INTERVAL: Time interval between terminal stdout status updates.
PRINT_INTERVAL = 0.01

# RESTART_FILE: Path to a .vts file to resume from. Leave blank if not restarting.
RESTART_FILE = 

# RESTART_TIME: Simulation time corresponding to the restart file.
RESTART_TIME = 0.0

[Probes]
# Define point probes to extract time-history data.
# Format: PROBE_NAME = X, Y, Variable
# Available variables: Density, XMomentum, YMomentum, Energy, Pressure, Temperature, Mach, Sigma
PROBE_1 = 0.5, 0.5, Pressure

PROBE_3 = 0.9, 0.5, Mach
